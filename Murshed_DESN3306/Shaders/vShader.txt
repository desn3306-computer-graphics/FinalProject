#version 330       
  
layout (location = 0) in vec3 pos;
layout (location = 1) in vec3 vCol;
layout (location = 2) in vec2 tex;
layout (location = 3) in vec3 norm;

uniform mat4 model;
uniform mat4 projModel;
uniform mat4 camera;

out vec4 vColour;
out vec2 TexCoord;
out vec3 Normal;
out vec3 FragPosition;


  

void main()   
{               
    gl_Position = projModel*camera*model*vec4(pos.x, pos.y, pos.z, 1.0);
    vColour=vec4(vCol, 1.0);
    TexCoord = tex; 
    Normal = mat3(transpose(inverse(model))) * norm;
    FragPosition = (model * vec4(pos, 1.0)).xyz;

}
